LOAD CSV WITH HEADERS FROM
'https://raw.githubusercontent.com/krishnaik06/graph-dataset/main/relationships.csv'
AS row
CALL (row) {
  MATCH (u1:User {userId: toInteger(row.userId1)})
  MATCH (u2:User {userId: toInteger(row.userId2)})
  MERGE (u1)-[:FRIEND]->(u2)
  MERGE (u1)-[:LIKES]->(u2)
} IN TRANSACTIONS OF 500 ROWS
;

# Some basic queries on these

# 1. Retrieve all users
MATCH (u:User) RETURN u

# 2. Retrieve all the posts
MATCH (p:Post) RETURN p

# 3. Retrive the friends of a specific userId
MATCH (u:User {name: "John"})-[:FRIEND]->(f:User)
RETURN f.name AS friendName

# 4. Retrieve the posts made by the friends of a particular user
MATCH (u: User{name:"John"})-[:FRIEND]->(f:User)-[:POSTED] ->(p:Post)
RETURN f.name AS friendName, p.content AS friendContent
;

# 5. Retrive all the likes of the posts of a particular user "John"
MATCH (u:User{name:"Jhon})-[:POSTED]->(p:Post)<-[:LIKES]-(l:User)
RETURN l.name AS friendName, p.content AS LikedContent
;

# 6. Count the number of friends each user have
MATCH(u:User)-[:FRIEND]-(f:User)
RETURN u.name, COUNT(f) AS numberOfFriends
ORDER BY numberOfFriends DESC
;
